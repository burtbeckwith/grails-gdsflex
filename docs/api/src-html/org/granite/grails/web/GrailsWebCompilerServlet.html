069        @Override
070        public void init(ServletConfig servletConfig) throws ServletException {
071            this.servletConfig = servletConfig;
072            this.servletContextLoader = new ServletContextResourceLoader(servletConfig.getServletContext());
073            ApplicationContext springContext = WebApplicationContextUtils.getRequiredWebApplicationContext(servletConfig.getServletContext());
074                    if (springContext.containsBean(GroovyPageResourceLoader.BEAN_ID))
075                            this.resourceLoader = (ResourceLoader)springContext.getBean(GroovyPageResourceLoader.BEAN_ID);
076                    
077            webCompiler = WebCompiler.getInstance();
078            webCompiler.setTargetPlayer("9.0.28");
079            
080            try {
081                webCompiler.init(servletConfig.getServletContext().getRealPath("/WEB-INF"));
082            } 
083            catch (Exception e) {
084                log.error(e);
085            }
086    
087            this.grailsAttributes = new DefaultGrailsApplicationAttributes(servletConfig.getServletContext());
088        }

092            doPost(request, response);

095        @Override
096        protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {        
097            WebCompilerType type = WebCompilerType.application;
098            
099            request.setAttribute(GrailsApplicationAttributes.REQUEST_SCOPE_ID, grailsAttributes);
100    
101            // Get the name of the Groovy script (intern the name so that we can
102            // lock on it)
103            String pageName = request.getServletPath();        
104            pageName = pageName.replaceAll("\\.swf$", ".mxml").intern();
105    
106            Resource requestedFile = getResourceForUri(pageName);
107            if (requestedFile == null) {
108                // log("\"" + pageName + "\" not found");
109                response.sendError(404, "\"" + pageName + "\" not found.");
110                return;
111            }
112            
113            String requestedPath = requestedFile.getFile().getAbsolutePath();
114            File mxmlFile = requestedFile.getFile();
115            File sourcePath = mxmlFile.getParentFile();
116            long sourcesLastModified = lastModified(sourcePath, new String[] { ".mxml", ".as" });
117            
118            String swfPath = requestedPath.substring(0, (requestedPath.length() - 5));
119            File swfFolder = new File(swfPath.substring(0, swfPath.lastIndexOf(File.separator)) + File.separator + "swf");
120            if (!swfFolder.exists())
121                    swfFolder.mkdir();
122            File swfFile = new File(swfFolder, swfPath.substring(swfPath.lastIndexOf(File.separator)+1) + ".swf");
123                    long lastModified = swfFile.exists() ? swfFile.lastModified() : 0;
124            
125            try {
126                if (mxmlFile != null && swfFile != null && lastModified < sourcesLastModified) {
127                                    // log("Compiling mxml file: " + mxmlFile.getAbsolutePath() + " to " + swfFile.getAbsolutePath());
128                    swfFile = webCompiler.compileMxmlFile(mxmlFile, swfFile, false, type, servletConfig.getServletContext().getContextPath());
129                            }
130            }
131            catch (WebCompilerException e) {
132                            if (swfFile == null || !swfFile.exists() || swfFile.lastModified() <= lastModified) {
133                                    PrintWriter writer = response.getWriter();
134                                    response.setContentType("text/html");
135                                    writer.println("<html><body>");
136                                    writer.println("<h1>Flex compilation error</h1>");
137                                    writer.println("<pre>");
138                                    writer.println(e.getMessage());
139                                    writer.println("</pre>");
140                                    writer.println("<p>Check server logs for more details</p>");
141                                    writer.println("</body></html>");
142                                    return;
143                            }
144            }
145                    
146                    File gspFile = new File(requestedPath.substring(0, (requestedPath.length() - 5)) + ".gsp");
147                    if (!gspFile.exists()) {
148                            File htmlFile = new File(swfFolder, swfPath.substring(swfPath.lastIndexOf(File.separator)+1) + ".html");
149                            if (htmlFile.exists())
150                                    htmlFile.renameTo(gspFile);
151                    }
152            
153            response.setContentType("application/x-shockwave-flash");
154            response.setContentLength((int)swfFile.length());
155            response.setBufferSize((int)swfFile.length());
156            response.setDateHeader("Expires", 0);
157            
158            OutputStream os = null;
159            InputStream is = null;
160            try {
161                is = new BufferedInputStream(new FileInputStream(swfFile));
162                os = response.getOutputStream();
163                for (int b = is.read(); b != -1; b = is.read())
164                    os.write(b);
165            } finally {
166                if (is != null) try {
167                    is.close();
168                } finally {
169                    if (os != null)
170                        os.close();
171                }
172            }
173        }

175        /**
176         * Attempts to retrieve a reference to a GSP as a Spring Resource instance for the given URI.
177         *
178         * @param uri The URI to check
179         * @return A Resource instance
180         */
181        public Resource getResourceForUri(String uri) {
182            Resource r = getResourceWithinContext(uri);
183            if (r != null && r.exists())
184                return r;
185            
186            // try plugin
187            String pluginUri = GrailsResourceUtils.WEB_INF + uri;
188            r = getResourceWithinContext(pluginUri);
189            if (r != null && r.exists())
190                return r;
191            
192            uri = getUriWithinGrailsViews(uri);
193            return getResourceWithinContext(uri);
194        }
195    
196        private Resource getResourceWithinContext(String uri) {
197            Resource r = servletContextLoader.getResource(uri);
198            if (r.exists()) 
199                return r;
200            return resourceLoader != null ? resourceLoader.getResource(uri) : null;
201        }
202        
203        /**
204         * Returns the path to the view of the relative URI within the Grails views directory
205         *
206         * @param relativeUri The relative URI
207         * @return The path of the URI within the Grails view directory
208         */
209        protected String getUriWithinGrailsViews(String relativeUri) {
210            StringBuffer buf = new StringBuffer();
211            String[] tokens;
212            if (relativeUri.startsWith("/"))
213                relativeUri = relativeUri.substring(1);
214    
215            if (relativeUri.indexOf('/') >= 0)
216                tokens = relativeUri.split("/");
217            else
218                tokens = new String[] { relativeUri };
219    
220            buf.append(GrailsApplicationAttributes.PATH_TO_VIEWS);
221            for (int i = 0; i < tokens.length; i++) {
222                String token = tokens[i];
223                buf.append('/').append(token);
224            }
225            
226            return buf.toString();
227        }

230        protected long lastModified(File path, final String[] extensions) {
231            long lastModified = 0L;
232            String[] files = path.list(new FilenameFilter() {
233                            @Override
234                            public boolean accept(File dir, String name) {
235                                    for (String ext : extensions) {
236                                            if (name.endsWith(ext))
237                                                    return true;
238                                    }
239                                    return false;
240                            }
241            });
242            for (String file : files) {
243                    File f = new File(path, file);
244                    if (f.isDirectory()) {
245                            long lm = lastModified(f, extensions);
246                            if (lm > lastModified)
247                                    lm = lastModified;
248                    }
249                    else if (f.lastModified() > lastModified)
250                            lastModified = f.lastModified();
251            }
252            
253            return lastModified;
254        }
